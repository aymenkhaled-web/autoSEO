-- KEYWORDS TRACKING
CREATE TABLE keywords (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id        UUID REFERENCES organizations(id) NOT NULL,
  site_id       UUID REFERENCES sites(id) NOT NULL,
  keyword       TEXT NOT NULL,
  search_volume INTEGER,
  difficulty    SMALLINT,
  cpc           NUMERIC(10,2),
  country       TEXT DEFAULT 'US',
  language      TEXT DEFAULT 'en',
  created_at    TIMESTAMPTZ DEFAULT now(),
  UNIQUE(site_id, keyword, country, language)
);
CREATE TABLE keyword_rankings (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  keyword_id    UUID REFERENCES keywords(id) NOT NULL,
  position      SMALLINT,
  url           TEXT,
  featured_snippet BOOLEAN DEFAULT false,
  checked_at    TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_keyword_rankings_kid_date ON keyword_rankings(keyword_id, checked_at DESC);
-- COMPETITORS
CREATE TABLE competitors (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id        UUID REFERENCES organizations(id) NOT NULL,
  site_id       UUID REFERENCES sites(id) NOT NULL,
  domain        TEXT NOT NULL,
  name          TEXT,
  last_analyzed_at TIMESTAMPTZ,
  created_at    TIMESTAMPTZ DEFAULT now()
);
-- BACKLINKS
CREATE TABLE backlinks (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id        UUID REFERENCES organizations(id) NOT NULL,
  site_id       UUID REFERENCES sites(id) NOT NULL,
  source_url    TEXT NOT NULL,
  target_url    TEXT NOT NULL,
  anchor_text   TEXT,
  domain_authority SMALLINT,
  is_dofollow   BOOLEAN,
  is_toxic      BOOLEAN DEFAULT false,
  first_seen_at TIMESTAMPTZ,
  last_seen_at  TIMESTAMPTZ,
  status        TEXT DEFAULT 'active' -- active, lost, disavowed
);
-- TEAM MEMBERS
CREATE TABLE team_members (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id        UUID REFERENCES organizations(id) NOT NULL,
  user_id       UUID REFERENCES users(id),
  email         TEXT NOT NULL,
  role          TEXT NOT NULL DEFAULT 'viewer', -- owner, admin, editor, viewer
  invited_by    UUID REFERENCES users(id),
  invite_token  TEXT,
  invite_status TEXT DEFAULT 'pending', -- pending, accepted, rejected
  invited_at    TIMESTAMPTZ DEFAULT now(),
  joined_at     TIMESTAMPTZ
);
-- ISSUE COMMENTS
CREATE TABLE issue_comments (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  issue_id      UUID REFERENCES issues(id) NOT NULL,
  org_id        UUID REFERENCES organizations(id) NOT NULL,
  user_id       UUID REFERENCES users(id) NOT NULL,
  content       TEXT NOT NULL,
  mentions      UUID[],
  created_at    TIMESTAMPTZ DEFAULT now()
);
-- ISSUE ASSIGNMENTS
ALTER TABLE issues ADD COLUMN assigned_to UUID REFERENCES users(id);
ALTER TABLE issues ADD COLUMN due_date TIMESTAMPTZ;
-- NOTIFICATIONS
CREATE TABLE notifications (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id        UUID REFERENCES organizations(id) NOT NULL,
  user_id       UUID REFERENCES users(id) NOT NULL,
  type          TEXT NOT NULL, -- issue_assigned, fix_applied, weekly_digest, etc
  title         TEXT NOT NULL,
  message       TEXT,
  link          TEXT,
  read          BOOLEAN DEFAULT false,
  created_at    TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_notif_user_unread ON notifications(user_id, read, created_at DESC);
-- NOTIFICATION PREFERENCES
CREATE TABLE notification_preferences (
  user_id       UUID PRIMARY KEY REFERENCES users(id),
  email_critical_issues BOOLEAN DEFAULT true,
  email_weekly_digest BOOLEAN DEFAULT true,
  email_fix_applied BOOLEAN DEFAULT true,
  email_assignments BOOLEAN DEFAULT true,
  slack_webhook_url TEXT,
  slack_critical_issues BOOLEAN DEFAULT false,
  in_app_all BOOLEAN DEFAULT true
);
-- SCHEDULED REPORTS
CREATE TABLE scheduled_reports (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id        UUID REFERENCES organizations(id) NOT NULL,
  site_id       UUID REFERENCES sites(id) NOT NULL,
  frequency     TEXT NOT NULL, -- daily, weekly, monthly
  recipients    TEXT[] NOT NULL,
  template      TEXT DEFAULT 'standard',
  next_run_at   TIMESTAMPTZ,
  enabled       BOOLEAN DEFAULT true,
  created_at    TIMESTAMPTZ DEFAULT now()
);
-- USAGE TRACKING
CREATE TABLE usage_events (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id        UUID REFERENCES organizations(id) NOT NULL,
  event_type    TEXT NOT NULL, -- crawl_pages, ai_fix, api_call, etc
  quantity      INTEGER DEFAULT 1,
  metadata      JSONB,
  created_at    TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_usage_org_type_date ON usage_events(org_id, event_type, created_at DESC);
-- WEBHOOKS
CREATE TABLE webhooks (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id        UUID REFERENCES organizations(id) NOT NULL,
  url           TEXT NOT NULL,
  events        TEXT[] NOT NULL, -- crawl_completed, fix_applied, issue_critical, etc
  secret        TEXT NOT NULL, -- for HMAC signature
  enabled       BOOLEAN DEFAULT true,
  last_triggered_at TIMESTAMPTZ,
  failure_count INTEGER DEFAULT 0,
  created_at    TIMESTAMPTZ DEFAULT now()
);
-- WEBHOOK DELIVERIES (for retry/debugging)
CREATE TABLE webhook_deliveries (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  webhook_id    UUID REFERENCES webhooks(id) NOT NULL,
  event_type    TEXT NOT NULL,
  payload       JSONB NOT NULL,
  status_code   SMALLINT,
  response_body TEXT,
  attempt       SMALLINT DEFAULT 1,
  delivered_at  TIMESTAMPTZ
);
-- AI USAGE TRACKING (cost monitoring)
CREATE TABLE ai_usage (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  org_id        UUID REFERENCES organizations(id) NOT NULL,
  model         TEXT NOT NULL, -- claude-sonnet-4-5, voyage-3
  input_tokens  INTEGER NOT NULL,
  output_tokens INTEGER NOT NULL,
  cost_cents    INTEGER NOT NULL,
  feature       TEXT, -- fix_generation, content_brief, etc
  created_at    TIMESTAMPTZ DEFAULT now()
);

```python
# packages/crawler/robots.py
import httpx
from urllib.robotparser import RobotFileParser
from urllib.parse import urljoin

async def get_robots_rules(domain: str, user_agent: str = "AutoSEO") -> RobotFileParser:
    robots_url = urljoin(domain, "/robots.txt")
    async with httpx.AsyncClient() as client:
        try:
            resp = await client.get(robots_url, timeout=10)
            parser = RobotFileParser()
            parser.parse(resp.text.splitlines())
            return parser
        except Exception:
            return RobotFileParser()  # No robots.txt = all allowed

def is_allowed(parser: RobotFileParser, url: str, user_agent: str = "AutoSEO") -> bool:
    return parser.can_fetch(user_agent, url)

def get_crawl_delay(parser: RobotFileParser, user_agent: str = "AutoSEO") -> float:
    delay = parser.crawl_delay(user_agent)
    return max(delay or 1.0, 1.0)
```

### Sitemap Parser

```python
# packages/crawler/sitemap.py
import httpx, gzip
from xml.etree import ElementTree
from typing import AsyncGenerator
from urllib.parse import urljoin

async def discover_sitemap_urls(domain: str) -> AsyncGenerator[str, None]:
    sitemap_urls = [
        urljoin(domain, "/sitemap.xml"),
        urljoin(domain, "/sitemap_index.xml"),
        urljoin(domain, "/sitemap-index.xml"),
    ]
    seen = set()
    for sitemap_url in sitemap_urls:
        async for url in _parse_sitemap(sitemap_url, seen):
            yield url

async def _parse_sitemap(url: str, seen: set) -> AsyncGenerator[str, None]:
    if url in seen: return
    seen.add(url)
    async with httpx.AsyncClient() as client:
        resp = await client.get(url, timeout=30, follow_redirects=True)
        if resp.status_code != 200: return
        content = resp.content
        if url.endswith(".gz"): content = gzip.decompress(content)
        root = ElementTree.fromstring(content)
        ns = {"sm": "http://www.sitemaps.org/schemas/sitemap/0.9"}
        for sitemap in root.findall("sm:sitemap/sm:loc", ns):
            async for u in _parse_sitemap(sitemap.text.strip(), seen): yield u
        for loc in root.findall("sm:url/sm:loc", ns):
            yield loc.text.strip()
```

```python
from crawl4ai import AsyncWebCrawler, CrawlerRunConfig, BrowserConfig
from crawl4ai.deep_crawling import BFSDeepCrawlStrategy
from crawl4ai.deep_crawling.filters import FilterChain, SEOFilter

async def crawl_with_crawl4ai(url: str, max_pages: int = 500) -> list[dict]:
    browser_config = BrowserConfig(
        browser_type="firefox", headless=True,
        use_managed_browser=True,
        extra_args=["--no-sandbox", "--disable-dev-shm-usage"]
    )
    crawl_config = CrawlerRunConfig(
        deep_crawl_strategy=BFSDeepCrawlStrategy(
            max_depth=5, max_pages=max_pages,
            include_external=False,
            filter_chain=FilterChain([SEOFilter(threshold=0.3)])
        ),
        mean_delay=1.5, max_range=0.5,
        word_count_threshold=10,
        exclude_external_links=True,
        wait_for="networkidle", page_timeout=30000,
    )
    pages = []
    async with AsyncWebCrawler(config=browser_config) as crawler:
        results = await crawler.arun(url=url, config=crawl_config)
        for result in results:
            if result.success:
                pages.append({"url": result.url, "html": result.html,
                              "markdown": result.markdown.raw_markdown,
                              "status_code": result.status_code, "source": "crawl4ai"})
    return pages
```

```python
    return any(s.lower() in html.lower() for s in BLOCKED_SIGNALS)

async def fetch_with_scrapfly(url: str) -> dict | None:
```

### SEO Signal Extractor

```python
# packages/crawler/extractor.py
from bs4 import BeautifulSoup
import json, re
from urllib.parse import urlparse

class SEOExtractor:
    def __init__(self, html: str, url: str):
        self.soup = BeautifulSoup(html, "lxml")
        self.url = url
        self.base_domain = urlparse(url).netloc

    def extract_all(self) -> dict:
        return {**self._meta(), **self._headings(), **self._links(),
                **self._images(), **self._schema(), **self._social(),
                **self._technical()}

    def _meta(self) -> dict:
        title = self.soup.find("title")
        desc = self.soup.find("meta", attrs={"name": "description"})
        canonical = self.soup.find("link", attrs={"rel": "canonical"})
        robots = self.soup.find("meta", attrs={"name": "robots"})
        t = title.get_text(strip=True) if title else None
        d = desc.get("content", "").strip() if desc else None
        return {"title": t, "title_length": len(t) if t else 0,
                "meta_description": d, "meta_description_length": len(d) if d else 0,
                "canonical_url": canonical.get("href") if canonical else None,
                "robots_directive": robots.get("content") if robots else "index, follow"}

    def _headings(self) -> dict:
        h1s = self.soup.find_all("h1")
        return {"h1_count": len(h1s),
                "h1_text": [h.get_text(strip=True) for h in h1s],
                "h2_count": len(self.soup.find_all("h2")),
                "h3_count": len(self.soup.find_all("h3"))}

    def _links(self) -> dict:
        links = self.soup.find_all("a", href=True)
        internal = [l for l in links if urlparse(l["href"]).netloc in ("", self.base_domain)]
        external = [l for l in links if urlparse(l["href"]).netloc not in ("", self.base_domain)]
        return {"internal_links_count": len(internal),
                "external_links_count": len(external)}

    def _images(self) -> dict:
        imgs = self.soup.find_all("img")
        return {"images_count": len(imgs),
                "images_missing_alt": sum(1 for i in imgs if not i.get("alt"))}

    def _schema(self) -> dict:
        schemas, types, errors = [], [], []
        for s in self.soup.find_all("script", type="application/ld+json"):
            try:
                data = json.loads(s.string or "")
                schemas.append(data)
                if "@type" in data: types.append(data["@type"])
            except json.JSONDecodeError as e:
                errors.append(str(e))
        return {"schema_types": types, "schema_valid": len(errors) == 0,
                "schema_errors": errors}

    def _social(self) -> dict:
        def og(p): t = self.soup.find("meta", attrs={"property": f"og:{p}"}); return t.get("content") if t else None
        def tw(n): t = self.soup.find("meta", attrs={"name": f"twitter:{n}"}); return t.get("content") if t else None
        return {"og_title": og("title"), "og_description": og("description"),
                "og_image": og("image"), "twitter_card": tw("card")}

    def _technical(self) -> dict:
        hreflang = self.soup.find_all("link", attrs={"rel": "alternate", "hreflang": True})
        return {"hreflang_tags": [{"lang": t.get("hreflang"), "href": t.get("href")} for t in hreflang],
                "word_count": len(self.soup.get_text().split())}
```

### Crawl Orchestrator

```python
# workers/tasks/crawl.py
from celery import shared_task
import asyncio

@shared_task(bind=True, max_retries=3)
def run_site_crawl(self, site_id: str, crawl_id: str):
    asyncio.run(_async_crawl(site_id, crawl_id))

async def _async_crawl(site_id: str, crawl_id: str):
    site = await db.get_site(site_id)
    await db.update_crawl(crawl_id, status="running", started_at=now())
    try:
        # Step 1: robots.txt
        robots = await get_robots_rules(site.domain)
        delay = get_crawl_delay(robots)
        # Step 2: sitemap seed
        sitemap_urls = [url async for url in discover_sitemap_urls(site.domain)]
        # Step 3: Select crawl method
        if site.connection_type in ("wordpress", "webflow", "shopify"):
            pages = await crawl_via_cms_api(site)
        else:
            pages = await crawl_with_tiered_crawler(site.domain, robots, site.crawl_max_pages)
        # Step 4: Extract SEO signals
        extracted = []
        for page_data in pages:
            signals = SEOExtractor(page_data["html"], page_data["url"]).extract_all()
            lighthouse = await get_lighthouse_data(page_data["url"])
            signals.update(lighthouse)
            signals["seo_score"] = calculate_page_score(signals)
            extracted.append(signals)
        # Step 5: Cross-page analysis
        extracted = detect_duplicates(extracted)
        extracted = analyze_internal_links(extracted)
        # Step 6: Save
        await db.bulk_insert_pages(crawl_id, site_id, extracted)
        issues = generate_issues(extracted)
        await db.bulk_insert_issues(crawl_id, site_id, issues)
        # Step 7: Trigger AI analysis
        run_ai_analysis.delay(crawl_id, site_id)
        # Step 8: Complete
        await db.update_crawl(crawl_id, status="completed",
            completed_at=now(), pages_crawled=len(extracted),
            issues_found=len(issues), seo_score=calculate_site_score(extracted))
    except Exception as e:
        await db.update_crawl(crawl_id, status="failed", error_message=str(e))
        raise
```

## CREDENTIAL ENCRYPTION

```python
# packages/shared/encryption.py
import os, base64, hmac, hashlib
from cryptography.hazmat.primitives.ciphers.aead import AESGCM

MASTER_KEY = bytes.fromhex(os.environ["DOPPLER_MASTER_ENCRYPTION_KEY"])

def derive_tenant_key(org_id: str) -> bytes:
    return hmac.new(MASTER_KEY, org_id.encode(), hashlib.sha256).digest()

def encrypt_credential(org_id: str, plaintext: str) -> tuple[str, str]:
    key = derive_tenant_key(org_id)
    nonce = os.urandom(12)
    ct = AESGCM(key).encrypt(nonce, plaintext.encode(), None)
    return base64.b64encode(ct).decode(), base64.b64encode(nonce).decode()

def decrypt_credential(org_id: str, ciphertext_b64: str, nonce_b64: str) -> str:
    key = derive_tenant_key(org_id)
    return AESGCM(key).decrypt(base64.b64decode(nonce_b64),
                               base64.b64decode(ciphertext_b64), None).decode()
```
